RUC / cédula jquery validador ecuador

de @macool

ejemplos

github: https://github.com/macool/cedula-ruc-ecuador-validator-jquery

uso básico

$("#cedula_1").validarCedulaEC();

con opciones

$("#cedula_2").validarCedulaEC({ the_classes: "color_blue", events: "blur" });

no validar a menos que hayan 10 o 13 caracteres

$("#cedula_4").validarCedulaEC({ strict: false });

con callbacks

$("#cedula_3").validarCedulaEC({ onValid: function () { window.alert("cédula válida."); console.log(this); }, onInvalid: function () { window.alert("cédula inválida."); console.log(this); } });
